Authorization Examples

JavaScript 
const actra = new Actra(schema, policy)

const decision = actra.evaluate({
  actor: { id: "user1", role: "admin" },
  action: "delete",
  resource: { owner: "user1" }
})

console.log(decision.allow)
Python 
from actra import Actra

# actra = Actra(schema, policy)

decision = actra.evaluate({
  "actor": { "id": "user1", "role": "admin" },
  "action": "delete",
  "resource": { "owner": "user1" }
})

print(decision.allow)
