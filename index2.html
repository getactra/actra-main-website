Control what agents can do — tools, APIs, and actions — in real time.

See it in action

JavaScript 
const actra = await Actra.create({
  schema,
  policy
})

const decision = await actra.evaluate({
  action: { type: "send_email" },
  actor: { role: "intern" }
})

// → { effect: "block", matched_rule: "no_external_email" }
Python 
from actra import Actra

actra = Actra.create(schema, policy)

decision = actra.evaluate({
  "action": {"type": "send_email"},
  "actor": {"role": "intern"}
})

# → {"effect": "block", "matched_rule": "no_external_email"}
